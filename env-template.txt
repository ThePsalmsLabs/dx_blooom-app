# IMPORTANT: Never commit .env.local to version control!
# This template file shows what's needed.

# =============================================================================
# PRIMARY RPC PROVIDERS (RECOMMENDED - START WITH THESE)
# =============================================================================

# Infura API Key (You already have this!)
# Get your free API key from: https://app.infura.io/
NEXT_PUBLIC_INFURA_API_KEY=your_infura_api_key_here

# Alchemy API Key (Highly recommended for Base chain)
# Get your free API key from: https://www.alchemy.com/
NEXT_PUBLIC_ALCHEMY_API_KEY=your_alchemy_api_key_here

# =============================================================================
# OPTIONAL PREMIUM RPC PROVIDERS (FOR MAXIMUM REDUNDANCY)
# =============================================================================

# QuickNode Endpoint (Premium service)
# Get your endpoint from: https://www.quicknode.com/
# NEXT_PUBLIC_QUICKNODE_URL=https://your-endpoint.quiknode.pro/your-project-id/

# Ankr API Key (Good free tier available)
# Get your free API key from: https://www.ankr.com/
# NEXT_PUBLIC_ANKR_API_KEY=your_ankr_api_key_here

# =============================================================================
# PRIVY AUTHENTICATION (REQUIRED - REPLACES RAINBOWKIT)
# =============================================================================

# Privy App ID (Required for wallet connection)
# Get your App ID from: https://console.privy.io/
NEXT_PUBLIC_PRIVY_APP_ID=your_privy_app_id_here

# =============================================================================
# BICONOMY SMART ACCOUNTS (OPTIONAL - FOR GASLESS TRANSACTIONS)
# =============================================================================

# Biconomy Paymaster API Key (For gas sponsorship)
# Get your API key from: https://dashboard.biconomy.io/
NEXT_PUBLIC_BICONOMY_PAYMASTER_API_KEY=your_biconomy_paymaster_api_key_here

# Biconomy Bundler URL (For transaction bundling)
# Get your bundler URL from: https://dashboard.biconomy.io/
NEXT_PUBLIC_BICONOMY_BUNDLER_URL=your_biconomy_bundler_url_here

# =============================================================================
# WALLET CONNECT (MOBILE WALLET SUPPORT - OPTIONAL)
# =============================================================================

# WalletConnect Project ID (Free)
# Get your project ID from: https://cloud.walletconnect.com/
# NEXT_PUBLIC_WALLETCONNECT_PROJECT_ID=your_walletconnect_project_id_here

# =============================================================================
# COINBASE INTEGRATION (OPTIONAL)
# =============================================================================

# Coinbase Project ID (For enhanced Coinbase Wallet features)
# Get your project ID from: https://developer.coinbase.com/
# NEXT_PUBLIC_COINBASE_PROJECT_ID=your_coinbase_project_id_here

# =============================================================================
# LEGACY SUPPORT (FOR EXISTING SETUPS)
# =============================================================================

# Legacy WalletConnect support (if you have existing setup)

# =============================================================================
# CONFIGURATION NOTES
# =============================================================================
#
# MIGRATION FROM RAINBOWKIT TO PRIVY:
# 1. REQUIRED: Add NEXT_PUBLIC_PRIVY_APP_ID (create account at https://console.privy.io/)
# 2. RECOMMENDED: Add Biconomy keys for gasless transactions
# 3. OPTIONAL: Keep your existing RPC providers (Infura, Alchemy)
#
# The Privy configuration will automatically:
# - Handle wallet connections with embedded wallets
# - Support email/SMS login in addition to wallet connections
# - Provide better user experience than RainbowKit
# - Integrate with your existing smart account setup
#
# After adding these variables:
# 1. Restart your development server
# 2. Test wallet connection with the new Privy interface
